	//Algorithm to surround bar (raw or column). Raw: excludeX < 0, column: excludeY < 0,
	private void surround(int x, int y, int excludeX, int excludeY, BiConsumer set){
		int sizeX = 0;
		int sizeY = 0;
		if(excludeX < 0)
		for (int i = x - 1; i < x + 1 + sizeX; i++) {
			for (int j = y - 1; j < y + 1 + sizeY; j++)
				if (x !=excludeX && y !=excludeY)
					set.accept(i, j);
		}
	}
